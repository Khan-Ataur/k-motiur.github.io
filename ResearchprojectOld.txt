 <script>
        const carousels = {};

        function showSlide(projectId, index) {
            const carousel = document.querySelector(`.carousel[data-project="${projectId}"]`);
            const slides = carousel.querySelectorAll('.slide');
            if (!carousels[projectId]) carousels[projectId] = 0;

            carousels[projectId] = (index + slides.length) % slides.length;

            slides.forEach((slide, i) => {
                slide.classList.toggle('active', i === carousels[projectId]);
            });
        }

        function nextSlide(projectId) {
            showSlide(projectId, carousels[projectId] + 1);
        }

        function prevSlide(projectId) {
            showSlide(projectId, carousels[projectId] - 1);
        }

        // Initialize all carousels
        document.querySelectorAll('.carousel').forEach((carousel, index) => {
            carousels[index + 1] = 0;
        });
    </script>
